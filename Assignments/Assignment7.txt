SELECT COUNT(*) AS OrderCount
FROM ORDERS
WHERE Odate = '1990-10-03';

2.
SELECT COUNT(DISTINCT City) AS CityCount
FROM CUSTOMERS
WHERE City IS NOT NULL;

3.
SELECT Cnum, MIN(Amt) AS SmallestOrder
FROM ORDERS
GROUP BY Cnum;

4.
SELECT Cname
FROM CUSTOMERS
WHERE Cname LIKE 'G%'
ORDER BY Cname ASC
LIMIT 1;

5.
SELECT City, MAX(Rating) 
FROM CUSTOMERS
GROUP BY City;

6.
SELECT Odate, COUNT(DISTINCT Snum) FROM ORDERS
GROUP BY Odate;
